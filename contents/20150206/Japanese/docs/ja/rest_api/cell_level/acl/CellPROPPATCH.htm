    xmlns:dc="urn:x-dc1:xmlns"  
    xmlns:Z="http://www.w3.com/standards/z39.50/"> 
    <D:set> 
        <D:prop> 
            <Z:Author>Author1 update</Z:Author> 
            <dc:hoge>fuga</dc:hoge> 
        </D:prop> 
    </D:set> 
    <D:remove> 
        <D:prop> 
            <Z:Author/> 
            <dc:hoge/> 
        </D:prop> 
    </D:remove> 
</D:propertyupdate>
